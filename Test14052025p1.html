import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

customers = pd.read_csv("customers.csv").copy()
customers_clean = customers.dropna(subset=['gender', 'nationallity', 'age']).copy()
customers_clean.loc[:, 'age'] = customers_clean['age'].astype(int)

gender_counts = customers_clean['gender'].value_counts()

plt.figure(figsize=(10, 6))
sns.barplot(x=gender_counts.index, y=gender_counts.values, palette="Spectral")

plt.title('devide by gender')
plt.xlabel('gender')
plt.ylabel('number of clients')
plt.grid(True)
plt.tight_layout()

plt.legend(loc=(1.1, 0))

plt.show()
